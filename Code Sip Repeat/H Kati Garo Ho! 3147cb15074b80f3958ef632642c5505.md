#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int, int> pii;
typedef pair<ll, ll> pll;

const ll INF = 1e18;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int test;
    cin >> test;
    
    while (test--) {
        int N, M;
        cin >> N >> M;
        
        vector<vector<pair<int, pii>>> adj(N + 1);
        vector<int> risks;
        
        for (int i = 0; i < M; i++) {
            int u, v, w, r;
            cin >> u >> v >> w >> r;
            adj[u].push_back({v, {w, r}});
            adj[v].push_back({u, {w, r}});
            risks.push_back(r);
        }
        
        // Sort and remove duplicate risks for binary search
        sort(risks.begin(), risks.end());
        risks.erase(unique(risks.begin(), risks.end()), risks.end());
        
        ll best_risk = INF, best_time = INF;
        
        // Binary search on the maximum allowed risk
        int left = 0, right = risks.size() - 1;
        while (left <= right) {
            int mid = (left + right) / 2;
            int max_allowed_risk = risks[mid];
            
            // Dijkstra to find minimum time with current risk constraint
            vector<ll> dist(N + 1, INF);
            dist[1] = 0;
            priority_queue<pll, vector<pll>, greater<pll>> pq;
            pq.push({0, 1});
            
            while (!pq.empty()) {
                auto [current_time, u] = pq.top();
                pq.pop();
                
                if (current_time != dist[u]) continue;
                
                for (auto& [v, edge] : adj[u]) {
                    int w = edge.first;
                    int r = edge.second;
                    
                    // Only use edges with risk <= max_allowed_risk
                    if (r <= max_allowed_risk && dist[v] > current_time + w) {
                        dist[v] = current_time + w;
                        pq.push({dist[v], v});
                    }
                }
            }
            
            if (dist[N] < INF) {
                // Found a path with current risk constraint
                best_risk = max_allowed_risk;
                best_time = dist[N];
                right = mid - 1; // Try to find smaller risk
            } else {
                left = mid + 1; // Need larger risk
            }
        }
        
        if (best_risk == INF) {
            cout << -1 << "\n";
        } else {
            // Now find the minimum time for the best risk
            // We need to run Dijkstra again with risk exactly <= best_risk
            // to ensure we get the minimum time
            vector<ll> dist(N + 1, INF);
            dist[1] = 0;
            priority_queue<pll, vector<pll>, greater<pll>> pq;
            pq.push({0, 1});
            
            while (!pq.empty()) {
                auto [current_time, u] = pq.top();
                pq.pop();
                
                if (current_time != dist[u]) continue;
                
                for (auto& [v, edge] : adj[u]) {
                    int w = edge.first;
                    int r = edge.second;
                    
                    if (r <= best_risk && dist[v] > current_time + w) {
                        dist[v] = current_time + w;
                        pq.push({dist[v], v});
                    }
                }
            }
            
            cout << best_risk << " " << dist[N] << "\n";
        }
    }
    
    return 0;
}
