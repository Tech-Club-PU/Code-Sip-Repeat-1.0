#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N, M;
    cin>>N>>M;

    vector<vector<int>> grid(N, vector<int>(M));
    vector<vector<int>> next(N, vector<int>(M));

    for (int i=0; i<N; i++)
        for (int j=0; j<M; j++)
            cin>>grid[i][j];

    // Directions for 8 neighbors
    int dx[]={-1,-1,-1,0,0,1,1,1};
    int dy[]={-1,0,1,-1,1,-1,0,1};

    for (int i=0; i<N; i++) {
        for (int j=0; j<M; j++) {
            int alive_neighbors=0;

            // Count alive neighbors
            for (int k=0; k<8; k++) {
                int ni=i+dx[k];
                int nj=j+dy[k];
                if (ni>=0 && ni<N && nj>=0 &&nj<M) {
                    alive_neighbors+=grid[ni][nj];
                }
            }

            // Apply rules
            if (grid[i][j]==1) {
                if (alive_neighbors == 2 || alive_neighbors == 3)
                    next[i][j]=1;
                else
                    next[i][j] = 0;
            } else { // dead cell
                if (alive_neighbors==3)
                    next[i][j]=1;
                else
                    next[i][j]=0;
            }
        }
    }

    // Print next state
    for (int i=0; i<N; i++) {
        for (int j=0; j<M; j++) {
            cout<<next[i][j];
            if (j<M-1) cout<<" ";
        }
        cout<<"\n";
    }

    return 0;
}
